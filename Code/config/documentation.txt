"Tile Name":{
		"typeOfTile": object being created - all implement tile interface
		"colour": colour of properties - null on all other
		"position": position displayed on board (currently 0-39) -1 from excel file
		"ownership": can a player own the tile
		"houseCost": cost to build a new level of house on a property - null on non-properties
		"cost": cost to buy tile - null on non-ownable
		"rent":{
			"0": rent for property with no housing
			"1": rent for property with 1 house - rent for station with 1 station owned - rent for utility with 1 utility owned
			"2": rent for property with 2 houses - rent for station with 2 stations owned - rent for utility with 2 utilities owned
			"3": rent for property with 3 houses - rent for station with 3 stations owned
			"4": rent for property with 4 houses - rent for station with 4 stations owned
			"5": rent for property with hotel
		},
		"behaviour": "function that is called when landed on/passed through for applicable"
	}
